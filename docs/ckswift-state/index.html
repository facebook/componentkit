State, state updates & bindings

Through the power of property wrappers, updating a state now simply requires changing its value (instead of calling -updateState:mode: and returning the new state from a block).

struct MyView : View {
  @State var shouldHideTooltip = false
  
  var body: Component {
    ComponentView<UIView>()
      .backgroundColor(.red)
  }
  
  private func didTapHideTooltipButton() {
    // Causes a state update on self
    shouldHideTooltip = true
  }
}

A binding can be also created to obtain a read/write proxy of an ancestor’s state (instead of passing the value as a prop + an action to update it):

struct MyView : View {
  @State var shouldHideTooltip = false
  
  var body: Component {
    MyChildView(shouldHideTooltip: $shouldHideTooltip)
  }
}
struct MyChildView : View {
  @Binding var shouldHideTooltip: Bool
  
  var body: Component {
    ComponentView<UIView>()
      .backgroundColor(.red)
  }
  private func didTapHideTooltipButton() {
    // Causes a state update on ancestor
    shouldHideTooltip = true
  }  
}
